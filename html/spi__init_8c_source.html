40/*
41 * @brief SPI intialization
42 * @param None
43 * @return API_SUCCESSS_test on success or AP_FAILURE_test on failure
44 */
45int spi_interface_init(void)
46{
47
48 int ret = 0;
49 int fd;
50
51 //Already Opened
52 if (spi_fd > 0)
53 {
54 close(spi_fd);
55 spi_fd = 0;
56 }
57
58 fd = open(spi_device, O_RDWR);
59 if (fd < 0)
60 {
61#ifdef init_error
62 printf("can't open SPI Flash device\n");
63#endif
64 return APP_SPI_INIT_ERROR;
65 }
66 else
67 {
68#ifdef init
69 printf("SPI open successful/n");
70#endif
71 }
72
73 //spi mode in mode 3 and (try for spidual mode if possible)
74 //spi_mode = SPI_CPHA | SPI_CPOL; //(1,1)=mode3 can also be (0,0) =mode0
75 spi_mode= SPI_MODE_0;
76
77 // spi mode
78 ret = ioctl(fd, SPI_IOC_WR_MODE32, &spi_mode);
79 if (ret == -1)
80 {
81#ifdef init_error
82 printf("can't set SPI mode\n");
83#endif
84 close(fd);
85 return APP_SPI_INIT_ERROR;
86 }
87 //bits per word
88 ret = ioctl(fd, SPI_IOC_WR_BITS_PER_WORD, &spi_bits);
89 if (ret == -1)
90 {
91#ifdef init_error
92 printf("can't set bits per word \n");
93#endif
94 close(fd);
95 return APP_SPI_INIT_ERROR;
96 }
97 //max speed hz
98 ret = ioctl(fd, SPI_IOC_WR_MAX_SPEED_HZ, &spi_speed);
99 if (ret == -1)
100 {
101#ifdef init_error
102 printf("can't set max speed hz \n");
103#endif
104 close(fd);
105 return APP_SPI_INIT_ERROR;
106 }
107
108 spi_fd = fd;
109 fd = 0;
110
111#ifdef init
112 printf("spi_fd :[%d]\n",spi_fd);
113 printf("spi mode: 0x%x\n", (uint32_t)spi_mode);
114 printf("bits per word: %d\n", spi_bits);
115 printf("max speed: %d Hz (%d KHz)\n", spi_speed, spi_speed/1000);
116#endif
117 return API_SUCCESSS;
118}

147/*
148 * @brief Exchange of the data through SPI
149 * @param spixfer pointer to the structure
150 * @return 0
151 */
152int SPI_Exchange(struct spi_flash_DataXfer *spixfer)
153{
154 int ret;
155 struct spi_ioc_transfer xfer[2];
156 memset(xfer, 0, sizeof xfer);
157 if(spixfer->Txlength)
158 {
159 xfer[0].tx_buf =(unsigned long) &spixfer->Txbuff[0];//framebuff;
160 xfer[0].rx_buf = (unsigned long)NULL;//(unsigned long)rx;
161 xfer[0].len = spixfer->Txlength; // Length of command to write
162 xfer[0].cs_change = 1; //0; // Keep CS activated
163 xfer[0].delay_usecs = spixfer->delay; //delay in us
164 xfer[0].speed_hz = spixfer->speed; //speed
165 xfer[0].bits_per_word = spixfer->bits; // bites per word 8
166 xfer[0].tx_nbits = 8;
167 }
168
169 if(spixfer->Txlength)
170 {
171 ret = ioctl(spixfer->fd, SPI_IOC_MESSAGE(1), &xfer);
172 }
173 if (ret < 1)
174 {
175 my_perror("can't send spi message\n");
176 return APP_SPI_INIT_ERROR;
177 }
178
179 return API_SUCCESSS;
180}

229 * @param data Data to be written
230 * @param datalen length of the data written
231 * @return API_SUCCESSS_test on completion
232 */
233int LCD_SPI_BUFFWRITE(uint8_t *data, uint16_t datalen)
234{
235 int ret;
236 int chucksize = 0,addroff;
237 struct spi_flash_DataXfer spi_flashDataXchange;
238
239
240 memset(&spi_flashDataXchange,0,sizeof(struct spi_flash_DataXfer));
241
242 if(datalen > (ST7735_SCREEN_WIDTH * ST7735_SCREEN_HEIGHT *2))
243 datalen = (ST7735_SCREEN_WIDTH * ST7735_SCREEN_HEIGHT *2);
244
245 addroff = 0;
246
247 while(datalen)
248 {
249 chucksize = (datalen > SPI_MAX_BUFFER_SIZE) ? SPI_MAX_BUFFER_SIZE : datalen;
250
251 spi_flashDataXchange.Txlength = chucksize;
252
253 // Copy data to the transmission buffer
254 memcpy(spi_flashDataXchange.Txbuff, &data[addroff], chucksize);
255
256 spi_flashDataXchange.fd=spi_fd;
257 spi_flashDataXchange.bits=spi_bits;
258 spi_flashDataXchange.speed=spi_speed;
259 spi_flashDataXchange.delay=0;
260
261 ret = SPI_Exchange(&spi_flashDataXchange);
262 if(ret)
263 {
264#ifdef init
265 printf("Returned from LCD_SPI_BUFFWRITE ret : %x\r\n",ret);
266#endif
267 return APP_SPI_INIT_ERROR;
268 }
269
270 addroff += chucksize;
271 datalen -= chucksize;
272 }
273
274 return API_SUCCESSS;
