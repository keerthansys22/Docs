523int spinlock_lock(pthread_spinlock_t *lock, unsigned int timeout_ms)
524{
525 struct timespec start, current;
526 clock_gettime(CLOCK_MONOTONIC, &start);
527
528 while (1) {
529 if (pthread_spin_trylock(lock) == 0) {
530 return RET_OK; // Lock acquired
531 }
532
533 clock_gettime(CLOCK_MONOTONIC, &current);
534 unsigned int elapsed_ms = (current.tv_sec - start.tv_sec) * 1000 +
535 (current.tv_nsec - start.tv_nsec) / 1000000;
536
537 if (elapsed_ms >= timeout_ms) {
538 return SPINLCK_TIMEOUT; // Timeout
539 }
540 }
541}
