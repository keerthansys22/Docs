227ssize_t receiveResponseFromApplication(mqd_t mq_id, ParamAccess_t *requestData)
228{
229 ssize_t bytes_received;
230 ssize_t MAX_MSG_SIZE = sizeof(ParamAccess_t);
231 // Call mq_receive
232 bytes_received = mq_receive(mq_id, (char *)requestData, MAX_MSG_SIZE, NULL);
233
234 if (bytes_received >= 0) {
235 // Success: Return the number of bytes received
236 return bytes_received;
237 }
238 else {
239 // Handle errors
240 switch (errno) {
241 case EAGAIN:
242 printf("Queue is empty (non-blocking mode)\n");
243 break;
244 case EBADF:
245 printf("Invalid message queue descriptor\n");
246 break;
247 case EINTR:
248 printf("Interrupted by a signal\n");
249 break;
250 case EMSGSIZE:
251 printf("Message buffer size is too small\n");
252 break;
253 case EINVAL:
254 printf("Invalid arguments provided\n");
255 break;
256 default:
257 printf("Unknown error occurred: %d\n", errno);
258 }
259 // Return failure indicator
260 return -1;
261 }
262
263}
