50uint8_t KeyPress_check(int i2cFile) {
51 uint8_t txBuffer[1];
52 uint8_t rxBuffer[2];
53
54//Writing the values
55 txBuffer[0] = 0x00;
56 i2cmsg.addr = I2C_ADDR2;
57 i2cmsg.flags = 0;
58 i2cmsg.len = sizeof(txBuffer[0]);
59 i2cmsg.buf = &txBuffer[0];
60
61 msg_rdwr.msgs = &i2cmsg;
62 msg_rdwr.nmsgs = 1;
63
64 if (ioctl(i2cFile, I2C_RDWR, &msg_rdwr) < 0)
65 {
66#ifdef selftest_error
67 my_perror("I2C write failed in keypress\r\n");
68#endif
69 return APP_I2C_INIT_ERROR;
70 }
71 //reading the valiue
72 i2cmsg.addr = I2C_ADDR2;
73 i2cmsg.flags = I2C_M_RD;
74 i2cmsg.len = sizeof(rxBuffer[0]);
75 i2cmsg.buf = &rxBuffer[0];
76
77 msg_rdwr.msgs = &i2cmsg;
78 msg_rdwr.nmsgs = 1;
79
80 if (ioctl(i2cFile, I2C_RDWR, &msg_rdwr) < 0) {
81#ifdef selftest_error
82 my_perror("I2C read failed in keypress\r\n");
83#endif
84 return APP_I2C_INIT_ERROR;
85 }
86
87 keypressed = rxBuffer[0] & 0x38;
88 if (keypressed == 0x38)
89 return APP_I2C_INIT_ERROR;
90
91 return API_SUCCESSS;
92}
