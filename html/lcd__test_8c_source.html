473/*
474 * @brief This function is to write the data through SPI
475 * @param data Data to be written
476 * @param datalen length of the data written
477 * @return API_SUCCESS_test on completion
478 */
479int LCD_SPI_BUFFWRITE1(uint8_t *data, uint16_t datalen)
480{
481 int ret;
482 int chucksize = 0,addroff;
483 struct spi_flash_DataXfer spi_flashDataXchange;
484
485
486 memset(&spi_flashDataXchange,0,sizeof(struct spi_flash_DataXfer));
487
488 if(datalen > (ST7735_SCREEN_WIDTH * ST7735_SCREEN_HEIGHT *2))
489 datalen = (ST7735_SCREEN_WIDTH * ST7735_SCREEN_HEIGHT *2);
490
491 addroff = 0;
492 while(datalen)
493 {
494 chucksize = (datalen > SPI_MAX_BUFFER_SIZE) ? SPI_MAX_BUFFER_SIZE : datalen;
495
496 spi_flashDataXchange.Txlength = chucksize;
497
498 // Copy data to the transmission buffer
499 memcpy(spi_flashDataXchange.Txbuff, &data[addroff], chucksize);
500
501 spi_flashDataXchange.fd=spi_fd;
502 spi_flashDataXchange.bits=spi_bits;
503 spi_flashDataXchange.speed=spi_speed;
504 spi_flashDataXchange.delay=0;
505
506
507
508
509 ret = SPI_Exchange(&spi_flashDataXchange);
510 if(ret)
511 {
512#ifdef selftest
513 printf("Returned from LCD_SPI_BUFFWRITE ret : %x\r\n",ret);
514#endif
515 return API_SUCCESS_test;
516 }
517
518 addroff += chucksize;
519 datalen -= chucksize;
520 }
521
522 return API_SUCCESS_test;
523
524}

574/*
575 * @brief Exchange of the data through SPI
576 * @param spixfer pointer to the structure
577 * @return 0
578 */
579
580int SPI_Exchange1(struct spi_flash_DataXfer *spixfer)
581{
582 int ret;
583 struct spi_ioc_transfer xfer[2];
584
585 memset(xfer, 0, sizeof xfer);
586
587 if(spixfer->Txlength)
588 {
589
590 xfer[0].tx_buf =(unsigned long) &spixfer->Txbuff[0];//framebuff;
591 xfer[0].rx_buf = (unsigned long)NULL;//(unsigned long)rx;
592 xfer[0].len = spixfer->Txlength; // Length of command to write
593 xfer[0].cs_change = 1; //0; // Keep CS activated
594 xfer[0].delay_usecs = spixfer->delay; //delay in us
595 xfer[0].speed_hz = spixfer->speed; //speed
596 xfer[0].bits_per_word = spixfer->bits; // bites per word 8
597 xfer[0].tx_nbits = 8;
598 }
599
600 if(spixfer->Txlength)
601 {
602 ret = ioctl(spixfer->fd, SPI_IOC_MESSAGE(1), &xfer);
603 }
604 if (ret < 1)
605 {
606 my_perror("can't send spi message\n");
607 return API_SUCCESS_test;
608 }
609
610 return API_SUCCESS_test;
611}
