178
179 for (;;)
180 {

183 printf("epoll_wait");
184 if (errno == EINTR)

186 }
187
188 for (n = 0; n < nfds; ++n)
189 {
190 if ((events[n].events & EPOLLERR) || (events[n].events & EPOLLHUP) || !(events[n].events & EPOLLIN)) {
191 printf("epoll error");
192 close(events[n].data.fd); // Close the file descriptor on error

196 if (events[n].data.fd == DATA_RECEIVING_EVENT_FD)

203 // Continue receiving messages until the queue is empty
