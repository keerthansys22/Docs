315uint8_t ST7291_setScreenWindow(uint16_t x1, uint16_t y1, uint16_t x2, uint16_t y2)
316{
317 uint8_t val;
318 /* Verify Coordinate ranges */
319 if ((x1 > x2) || (x2 > ST7735_SCREEN_WIDTH) ||
320 (y1 > y2) || (y2 > ST7735_SCREEN_HEIGHT))
321 {
322 // out of range
323 return 1;
324 }
325
326 /* Set LCD Coloumn*/
327 LCD_SPI_CMDWRITE(ST7291_CASET); /* column address set */
328 val = (uint8_t)(x1 >> 8);
329 LCD_SPI_DATAWRITE(val); /* send start x position */
330 LCD_SPI_DATAWRITE((uint8_t)x1);
331 val = (uint8_t)(x2 >> 8);
332 LCD_SPI_DATAWRITE(val); /* send end x position */
333 LCD_SPI_DATAWRITE((uint8_t)x2);
334 /* Set LCD Row*/
335 LCD_SPI_CMDWRITE(ST7291_RASET);
336 val = (int)(y1 >> 8);
337 LCD_SPI_DATAWRITE(val); /* send start y position */
338 LCD_SPI_DATAWRITE((uint8_t)y1);
339
340 val = (uint8_t)(y2 >> 8);
341 LCD_SPI_DATAWRITE(val);
342 LCD_SPI_DATAWRITE((uint8_t)y2);/*send end x position*/
343 LCD_SPI_CMDWRITE(ST7291_RAMWR); /* Access RAM */
344
345 return 0;
346}

388void lcd_refresh(lv_color_t *framebuf)
389{
390 uint32_t j, pixcelNum = ST7735_SCREEN_WIDTH * ST7735_SCREEN_HEIGHT;
391 uint32_t pixval;
392
393 ST7291_setScreenWindow(0, 0, ST7735_SCREEN_WIDTH, ST7735_SCREEN_HEIGHT);
394
395 for (j = 0; j < pixcelNum; j++)
396 {
397 pixval = framebuf->full;
398 LCD_SPI_DATAWRITE((uint8_t)((pixval >> 16) & 0xFF)); // R
399 LCD_SPI_DATAWRITE((uint8_t)((pixval >> 8) & 0xFF)); // G
400 LCD_SPI_DATAWRITE((uint8_t)(pixval & 0xFF)); // G
401 framebuf++;
402 }
403}

410void lcd_refreshbuf(lv_color_t *framebuf)
411{
412
413 uint32_t pixval;
414 uint8_t *pixbuf = NULL;
415
416 pixbuf = (uint8_t *)malloc(ST7735_SCREEN_WIDTH*ST7735_SCREEN_HEIGHT * 3); // 3 bytes for RGB666
417
418 for (int j = 0; j < (ST7735_SCREEN_WIDTH* ST7735_SCREEN_HEIGHT * 3);)
419 {
420 pixval = framebuf->full;
421 pixbuf[j++] = (uint8_t)((pixval >> 16) & 0xFF); // R
422 pixbuf[j++] = (uint8_t)((pixval >> 8) & 0xFF); // G
423 pixbuf[j++] = (uint8_t)(pixval & 0xFF); // G
424 framebuf++;
425 }
426
427 ST7291_setScreenWindow(0, 0, ST7735_SCREEN_WIDTH, ST7735_SCREEN_HEIGHT);
428
429 LCD_SET_DATACMD(1);
430 LCD_SPI_BUFFWRITE(&pixbuf[0], (ST7735_SCREEN_WIDTH * ST7735_SCREEN_HEIGHT * 3)); // 3 bytes for RGB666
431
432 free(pixbuf);
433
434}
